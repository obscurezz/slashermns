-- User functions list

-- все юниты игрока, наносящие урон, имеют ограничение при выборе цели
-- RangeRestriction.NoRestriction           - нет дополнительных ограничений
-- RangeRestriction.NearestLine             - можно выбирать цели только из ближайшего ряда
-- RangeRestriction.NearestLineIfSecondLine - можно выбирать цели только из ближайшего ряда, если атакующий находится во втором ряду
set_damageUnitsRangeRestriction(value)

-- в бою против игрока юниты нейтральных отрядов выбирают цели случайно
-- true - включить настройку
-- false - выключить настройку
set_neutralsSelectTargetsRandomly(value)

-- Обычный урон, вампиризм и тауматургия первичных атак юнитов (но не предметов) 
-- начинают наносить урон с разбросом (в зависимости от базовой точности юнита),
-- но точность данных атак становится 100%
-- Варианты разброса:
-- SpreadMode.NoSpread  - оставить как в оригинале
-- SpreadMode.Smooth    - плавное изменение от минимального до максимального урона
-- SpreadMode.MinMax    - урон или минимальный, или максимальный
-- SpreadMode.MinMidMax - урон или минимальный, или максимальный, или средний
-----------------------------
-- Бафы и дебафы точности не влияют на шанс промаха, но влияют на удачу.
-- У юнита базовая удача 100, ее можно снизить до 0 или увеличить до 200 модификаторами точности.
-- Чем выше удача, тем больше в среднем будет наноситься урона.
-- При удаче, равной 0, урон всегда минимальный.
-- При удаче, равной 200, урон всегда максимальный.
set_damageSpreadMode(value)

-- На величину разброса будет влиять точность (см. ниже)
-- true - включить настройку
-- false - выключить настройку
-- по умолчанию true
set_damageSpreadApplyPower(value)

-- Масштабирующий множитель для максимального урона (value)
-- По умолчанию 1.0
-- Средний_урон:
-- базовый_урон * базовая_точность / 100
-- Максимальный урон:
-- при set_damageSpreadApplyPower(true)
-- Средний_урон * ( 1 + value * (100 - базовая_точность) / 100)
-- при set_damageSpreadApplyPower(false)
-- Средний_урон * ( 1 + value)
set_damageSpreadUpperMulti(value)

-- Масштабирующий множитель для минимального урона (value)
-- По умолчанию 1.0
-- Средний_урон:
-- базовый_урон * базовая_точность / 100
-- Минимальный урон:
-- при set_damageSpreadApplyPower(true)
-- Средний_урон * ( 1 - value * (100 - базовая_точность) / 100)
-- при set_damageSpreadApplyPower(false)
-- Средний_урон * ( 1 - value)
set_damageSpreadLowerMulti(value)

-- Разброс для Лечения
-- Варианты разброса:
-- SpreadMode.NoSpread  - оставить как в оригинале
-- SpreadMode.Smooth    - плавное изменение от минимального до максимального урона
-- SpreadMode.MinMax    - личение или минимальное, или максимальное
-- SpreadMode.MinMidMax - личение или минимальное, или максимальное, или среднее
set_healSpreadMode(value)

-- Масштабирующий множитель для максимального лечения (value)
-- По умолчанию 0.2
-- Максимального лечения:
-- базовое_лечение * ( 1 + value )
set_healSpreadUpperMulti(value)

-- Масштабирующий множитель для минимального лечения (value)
-- По умолчанию 0.2
-- Минимального лечения:
-- базовое_лечение * ( 1 - value )
set_healSpreadLowerMulti(value)

-- Шанс полноценного срабатывания лечения
-- от 0 до 100
-- в статах юнита все еще будет написано 100 точности
-- не зависит от бафов и дебафов
set_healHitChance(value)

-- На это число будет умножена величина лечения при "промахе"
-- по умолчанию 0
set_healMissMulti(value)

-- баферы под управлением игрока становятся дебаферами
-- true - включить настройку
-- false - выключить настройку
set_playersBuffersAreDebuffers(value)

-- дебаферы под управлением игрока становятся баферами
-- true - включить настройку
-- false - выключить настройку
set_playersDebuffersAreBuffers(value)
