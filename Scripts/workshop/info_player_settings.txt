-- User functions list

--[[
	Комментарий ко всем функициям ниже, у которых есть параметр (leaderType):
	Можно указать, на отряды под чьим предводительством будет распространятся бонус
	Если не указывать ничего, то бонус будет применяться ко всем отрядам выбранного игрока (эквивалентно GroupLeaderType.Any)
	
	Список типов (например, чтобы указать отряды с Магом во главе, нужно написать GroupLeaderType.Mage вместо leaderType):
	Any      - все отряды игрока
	Nobody   - отряды без лидера (руины или гарнизон замка)
	Fighter  - отряды под предводительством Воина
	Explorer - отряды под предводительством Разведчика
	Mage     - отряды под предводительством Мага
	Rod      - отряды под предводительством Жезловика
	Noble    - отряды под предводительством Вора
	Summon   - призываемое на глобальную карту Cущество
	Illusion - призываемая на глобальную карту Иллюзия
	Guardian - столица со Стражем
	
	Пример:
	Доступ к бонусам, которые будут действовать на все отряды игрока, находящиеся на чужой земле:
	get_otherTerrainStatsBonus()
	или
	get_otherTerrainStatsBonus(GroupLeaderType.Any)
	
	Доступ к бонусам, которые будут действовать на все отряды игрока под предводительством лидера-разведчика, находящиеся на чужой земле:
	get_otherTerrainStatsBonus(GroupLeaderType.Explorer)
--]]
--[[
	Вот эти бонусы между собой складываются, при этом если указать какой-то бонус для всех отрядов 
	и для какого-то типа предводителя, то они тоже складываются:
	get_constantStatsBonus
	get_yourTerrainStatsBonus
	get_opponentTerrainStatsBonus
	get_neutralTerrainStatsBonus
	get_perLostHPStatsBonus
	get_perDifferentUnitStatsBonus
	get_perIdenticalUnitStatsBonus
	
	Для get_specialSettings другая механика взаимодействия в случае, 
	если что-то задано и для всех отрядов, и для типа предводителя:
	Более высокий приоритет имеет то значение настройки, которое выставлено для типа предводителя	
--]]
--[[
	Список рас (например, чтобы указать Эльфов, нужно написать Race.Elf вместо raceType):
	Human   - империя
	Undead  - нежить
	Heretic - легионы
	Dwarf   - кланы
	Elf     - эльфы
	Neutral - нейтралы
	
	Пример:
	Установить в качестве расы Нейтралов:
	set_race(Race.Neutral)
--]]

-- установить расу, для которой будут применены настройки
set_race(raceType)

-- установить земли рас, которые считаются противниками
-- в фигурных скобках через запятую нужные расы
set_opponentTerrain({raceType1, raceType2, ...})

-- установить земли рас, которые считаются нейтральными
-- в фигурных скобках через запятую нужные расы
set_neutralTerrain({raceType1, raceType2, ...})

-- получить доступ к бонусам, которые будут действовать всегда
-- полный список бонусов, доступных здесь, можно посмотреть в info_unit_stats_settings.txt
get_constantStatsBonus(leaderType)

-- получить доступ к бонусам, которые будут действовать на родной земле
-- полный список бонусов, доступных здесь, можно посмотреть в info_unit_stats_settings.txt
get_yourTerrainStatsBonus(leaderType)

-- получить доступ к бонусам, которые будут действовать на земле оппонента
-- полный список бонусов, доступных здесь, можно посмотреть в info_unit_stats_settings.txt
get_opponentTerrainStatsBonus(leaderType)

-- получить доступ к бонусам, которые будут действовать на нейтральной земле
-- полный список бонусов, доступных здесь, можно посмотреть в info_unit_stats_settings.txt
get_neutralTerrainStatsBonus(leaderType)

-- получить доступ к бонусам, которые будут действовать в зависимости от потери здоровья юнитом
-- заданная величина бонуса будет умножаться на 100*(максимальное_здоровье - текущее_здоровье)/максимальное_здоровье
-- полный список бонусов, доступных здесь, можно посмотреть в info_unit_stats_settings.txt
get_perLostHPStatsBonus(leaderType)

-- получить доступ к бонусам, величина которых зависит от количества различающихся юнитов в отряде
-- (юнит-лидер и не лидер это разные юниты)
-- полный список бонусов, доступных здесь, можно посмотреть в info_unit_stats_settings.txt
get_perDifferentUnitStatsBonus(leaderType)

-- получить доступ к бонусам, величина которых зависит от количества повторяющихся юнитов в отряде
-- (юнит-лидер и не лидер это разные юниты)
-- полный список бонусов, доступных здесь, можно посмотреть в info_unit_stats_settings.txt
get_perIdenticalUnitStatsBonus(leaderType)

-- получить доступ к особым настройкам
-- полный список можно посмотреть в info_special_settings.txt
get_specialSettings(leaderType)
